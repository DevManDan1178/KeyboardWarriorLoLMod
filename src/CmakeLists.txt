add_executable(KeyboardWarriorLoLMod
    main.cpp
    LoLReader.cpp
    HotkeyManager.cpp
    Messages.cpp
    LoLEventHandler.cpp
    InputReader.cpp
    ChatSender.cpp
    ui/MessagesUI.cpp
    ui/HotkeysUI.cpp
    ui/CoreUI.cpp
    helpers/HTTPHelper.cpp
)

set_target_properties(KeyboardWarriorLoLMod PROPERTIES
    WIN32_EXECUTABLE $<CONFIG:Release>
)

add_custom_command(TARGET KeyboardWarriorLoLMod POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory
        "$<TARGET_FILE_DIR:KeyboardWarriorLoLMod>/config"

    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${CMAKE_SOURCE_DIR}/config"
        "$<TARGET_FILE_DIR:KeyboardWarriorLoLMod>/config"
)


target_link_libraries(KeyboardWarriorLoLMod PRIVATE
    SDL2::SDL2
    uiohook
    opengl32
)

target_sources(KeyboardWarriorLoLMod PRIVATE
    ../external/imgui/imgui.cpp
    ../external/imgui/imgui_draw.cpp
    ../external/imgui/imgui_tables.cpp
    ../external/imgui/imgui_widgets.cpp
    ../external/imgui/imgui_demo.cpp
    ../external/imgui/backends/imgui_impl_sdl2.cpp
    ../external/imgui/backends/imgui_impl_opengl3.cpp
)

target_include_directories(KeyboardWarriorLoLMod PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/ui
    ${CMAKE_CURRENT_SOURCE_DIR}/helpers
    ../external/imgui
    ../external/imgui/backends
    ../external/httplib
)

# ---- install exe ----
install(TARGETS KeyboardWarriorLoLMod
        RUNTIME DESTINATION bin)


# ---- copy SDL DLL next to exe (dev convenience) ----
add_custom_command(TARGET KeyboardWarriorLoLMod POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        $<TARGET_FILE:SDL2::SDL2>
        $<TARGET_FILE_DIR:KeyboardWarriorLoLMod>
)
